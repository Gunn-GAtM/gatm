\section{Your Daily Dose of Vitamin $i$}

\newcounter{vit_i_problem_ii}

\begin{enumerate}
\item We will use complex numbers to find identities for $\cot$. Use Pascal's triangle to expand the following:
\begin{enumerate}
\begin{multicols}{3}
\item $(a+b)^3$
\item $(a+b)^4$
\item $(a+b)^5$
\setcounter{vit_i_problem_ii}{\value{enumii}}
\end{multicols}
\end{enumerate}
Then substitute $b=i=\sqrt{-1}$ and expand:
\begin{enumerate}
\begin{multicols}{3}
\setcounter{enumii}{\value{vit_i_problem_ii}}
\item $(a+i)^3$ %compare-books-disable
\item $(a+i)^4$ %compare-books-disable
\item $(a+i)^5$ %compare-books-disable
\setcounter{vit_i_problem_ii}{\value{enumii}}
\end{multicols}
\end{enumerate}
Finally, substitute $a=\cot \theta$ and expand:
\begin{enumerate}
\begin{multicols}{3}
\setcounter{enumii}{\value{vit_i_problem_ii}}
\item $(\cot \theta +i)^3$ %compare-books-disable
\item $(\cot \theta +i)^4$ %compare-books-disable
\item $(\cot \theta +i)^5$ %compare-books-disable
\setcounter{vit_i_problem_ii}{\value{enumii}}
\end{multicols}
\end{enumerate}
Consider $z=i+\cot\theta$.
\begin{enumerate}
\setcounter{enumii}{\value{vit_i_problem_ii}}
\item Use the above results to find identities for (i) $\cot 3\theta$, (ii) $\cot 4\theta$, and (iii) $\cot 5\theta$. %compare-books-disable
\item Graph $z$, $z^2$, $z^3$, $z^4$, and $z^5$, with $\theta \approx 75^\circ$. What method did you use? %compare-books-disable
\end{enumerate}

\item Compute $(1+i)^n$ for $n=3,4,5,\ldots$. Can you find a general pattern? %compare-books-disable
\item Expand and graph $\cis^n \theta$ for $n=2,3,4,\ldots$ and $\theta\approx 50^\circ$. %compare-books-disable

\begin{enumerate}
\item Why is the real part $\cos n\theta$ and the imaginary part $\sin n\theta$? %compare-books-disable
\item Use your results to write identities for $\cos n\theta$ and $\sin n\theta$ for $n=2,3,4,5$. %compare-books-disable
\end{enumerate}
